### importing params from terminal
ros2 run my_py_pkg number_publisher --ros-args -p number:=3 --ros-args -p timer_period:=0.5

### importing params from yaml file
ros2 run my_py_pkg number_publisher --ros-args --params-file ~/yaml_params/numer_params.yaml

### importing params from multu-params yaml file
ros2 run my_py_pkg number_publisher --ros-args -r __node:=number_publisher1 --params-file ~/yaml_params/number_params.yaml

### show topic
ros2 topic echo /number
ros2 topic hz /number

ros2 param set /number_publisher number 4
